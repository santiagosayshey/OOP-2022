the program has a function "sum_positive" that takes in an array and an int "n"
iterates through the array "n" times
determines if the current element is greater than 0 
sums those repsective numbers together
sum is returned to main function

the main function provides test cases


// include standard i/o libraries
#include <iostream>

// Sums together all positive numbers and returns them
// function sum_positive returns an int, takes in an array of ints and an int "n"
int sum_positive(int array[], int n) { 
    
    // initialise total
    int total = 0; 

    // iterate through the array n times at a pace of 1
    for (int i = 0; i < n; i++) { 

        // check for positive ints
        if (array[i] > 0) {
            
            // positives to total
            total += array[i];
        }
    }

    // return the total
    return total;
}

// main function
int main() {
    
    assigns an array of test numbers
    int test_array[5] = {5,3,4,-1,8};
    std::cout << sum_positive(test_array, 5) << std::endl;
    return 0;
}
